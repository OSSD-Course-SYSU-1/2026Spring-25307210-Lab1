// 导入OpenHarmony官方Hvigor插件核心任务
import { appTasks } from '@ohos/hvigor-ohos-plugin';
// 导入OAT审计插件（假设已封装为npm包，用于集成OAT扫描到构建流程）
import { oatAuditPlugin } from '@ohos/hvigor-oat-plugin';
// 导入自定义构建任务工具类
import { customBuildUtils } from './build/customBuildUtils';

/**
 * Hvigor构建配置入口
 * 功能：1. 基础构建任务；2. 集成OAT开源审计；3. 自定义构建钩子
 */
export default {
  // 核心构建任务：使用OpenHarmony官方内置插件（不可修改）
  system: appTasks,
  /**
   * 自定义插件列表：扩展Hvigor功能
   * 1. oatAuditPlugin：集成OAT开源审计扫描
   * 2. 自定义插件：可添加构建校验、产物处理等插件
   */
  plugins: [
    // 配置OAT审计插件，指定配置文件路径和扫描触发时机
    oatAuditPlugin({
      configPath: './OAT.xml', // 关联升级后的OAT配置文件
      triggerOn: ['build', 'assemble'], // 构建/打包时触发审计
      failOnError: true, // 审计失败时终止构建
      reportPath: './build/reports/oat-audit-result.html' // 审计报告输出路径
    })
  ],
  /**
   * 自定义构建任务：扩展工程构建能力
   */
  tasks: {
    // 自定义任务：构建前执行OAT预扫描
    preOatAudit: {
      // 任务依赖：执行前先完成clean任务
      dependsOn: ['clean'],
      // 任务执行逻辑
      action: () => {
        console.log('Starting pre-build OAT audit scan...');
        // 调用工具类执行预扫描
        customBuildUtils.runOatPreScan('./OAT.xml');
        console.log('Pre-build OAT audit scan completed!');
      }
    },
    // 自定义任务：构建后生成审计报告
    postOatReport: {
      // 任务依赖：构建完成后执行
      dependsOn: ['build'],
      // 任务执行逻辑
      action: () => {
        console.log('Generating OAT audit report...');
        customBuildUtils.generateOatReport('./build/reports/oat-audit-result.html');
        console.log('OAT audit report generated at ./build/reports/');
      }
    }
  },
  /**
   * 构建钩子：全局构建流程拦截
   */
  hooks: {
    // 构建开始前钩子：执行预扫描
    beforeBuild: (context) => {
      context.logger.info('=== Before Build: Run OAT pre-audit ===');
      context.tasks.preOatAudit.execute();
    },
    // 构建完成后钩子：生成报告
    afterBuild: (context) => {
      context.logger.info('=== After Build: Generate OAT report ===');
      context.tasks.postOatReport.execute();
    }
  }
}
